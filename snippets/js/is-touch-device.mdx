Verify if the current user's device is a touchscreen

<Code>
  {`/**
* Verify if the current browser is a touch device
* @returns {boolean}
*/
const isTouchDevice = () => (('ontouchstart' in window)
    || (navigator.maxTouchPoints > 0)
    || (navigator.msMaxTouchPoints > 0));`}
</Code>
